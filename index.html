async function handleChat(e) { if (e.key === 'Enter') { const input = document.getElementById('userInput'); const messagesDiv = document.getElementById('messages'); const userText = input.value; messagesDiv.innerHTML += `

You: ${userText}

`; input.value = ''; try { const response = await fetch('/api/chat', { method: 'POST', body: JSON.stringify({ message: userText }) }); const data = await response.json(); messagesDiv.innerHTML += `

AI: ${data.reply}

`; } catch (err) { messagesDiv.innerHTML += `

Error: Could not reach the AI.

`; } } }
