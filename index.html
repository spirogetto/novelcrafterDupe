async function handleChat(e) { if (e.key === 'Enter' && e.target.value.trim() !== '') { const input = document.getElementById('userInput'); const messagesDiv = document.getElementById('messages'); const userText = input.value; // 1. Display User Message messagesDiv.innerHTML += `
You: ${userText}
`; input.value = ''; // Clear input immediately // 2. Add a temporary "Loading" indicator const loadingId = "loading-" + Date.now(); messagesDiv.innerHTML += `
AI is thinking...
`; // Auto-scroll to bottom messagesDiv.scrollTop = messagesDiv.scrollHeight; try { const response = await fetch('/api/chat', { method: 'POST', body: JSON.stringify({ message: userText }) }); const data = await response.json(); // 3. Remove loading indicator and show actual reply document.getElementById(loadingId).remove(); messagesDiv.innerHTML += `
AI: ${data.reply}
`; } catch (err) { document.getElementById(loadingId).remove(); messagesDiv.innerHTML += `
Error: Could not reach the AI.
`; } // Final Auto-scroll messagesDiv.scrollTop = messagesDiv.scrollHeight; } }
